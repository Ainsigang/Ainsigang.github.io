linux makefile文件编写

hello:hello.o #生成hello可执行文件依赖于hello.o文件
gcc -o hello hello.o #生成hello可执行文件的命令
hello.o:hello.c #生成hello.o文件依赖于hello.c文件
gcc -c hello.c #编译生成hello.o文件
.PHONY:clean #定义一个伪目标用于清除执行过程中生成的中间文件
clean:
rm *.o #清除中间文件hello.o

make #编译生成可执行文件
make clean #清除中间文件即hello.o文件
```
